Features

import { flowbase } from '@flowbase/sdk'

const dashboard = await flowbase.analytics.create({
  metrics: ['velocity', 'health', 'kpis'],
  realtime: true
})

dashboard.onUpdate((data) => {
  console.log(data.velocity) // live updates
})
const doc = await flowbase.docs.create({
  title: 'Sprint Planning',
  collaborators: ['@team'],
  versionHistory: true
})

// Real-time editing with comments
doc.onEdit((change) => sync(change))
const workflow = flowbase.automate.build({
  trigger: 'task.completed',
  actions: [
    { type: 'notify.slack', channel: '#team' },
    { type: 'update.analytics' }
  ]
})

workflow.run() // no code required
